top

top命令之后输入1，可以列出多核CPU信息

[root@monitor ~]# top
top - 13:53:13 up 51 days, 19:11,  1 user,  load average: 0.00, 0.00, 0.00
Tasks: 140 total,   1 running, 139 sleeping,   0 stopped,   0 zombie
Cpu0  :  0.7%us,  1.0%sy,  0.0%ni, 98.3%id,  0.0%wa,  0.0%hi,  0.0%si,  0.0%st
Cpu1  :  0.3%us,  0.7%sy,  0.0%ni, 99.0%id,  0.0%wa,  0.0%hi,  0.0%si,  0.0%st
Mem:   3921112k total,  3705252k used,   215860k free,   191864k buffers
Swap:        0k total,        0k used,        0k free,  2154612k cached

其中比较重要的信息有load average，mem, cpu，load average后面分别是1分钟、5分钟、15分钟的负载情况，一般而言load 的值超过CPU核数的值就代表服务器已经超负荷工作，此时就需要进一步排查程序阻塞的原因，从而判断是需要添加机器还是优化程序

free -m
vmstat -m

查看内存信息

cat /proc/meminfo

查看Slab占用,有些时候内存占用无法通过top命令查看到占用的进程,这个时候就需要进一步排查innode slab之类的系统buffer cache 文章http://huoding.com/2015/06/10/444

slabtop -s c | head

##strace 排查进程
如果需要查看进程的调用栈,可以使用下面的命令,可以看到很多信息

strace -Ttt -s 1024 -p 888
strace -f -e trace=open,stat,close,unlink -p $(ps aux | grep php |awk '{print $2}' | xargs | tr ' ' ',')

iostat

查看系统IO读写

iostat 1

查看TIME_WAIT

netstat -n | awk '/^tcp/ {++S[$NF]} END {for(a in S) print a, S[a]}'

如果TIME_WAIT堆积过多，则需要优化系统内核参数配置 vim /etc/sysctl.conf，在底部添加保存，然后执行sysctl -p

net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_tw_recycle = 1
net.ipv4.tcp_fin_timeout = 30

其他优化参数

net.ipv4.tcp_keepalive_time = 1200
net.ipv4.ip_local_port_range = 10000 65000

查看TIME_WAIT占用内存

slabtop -o | grep tw

查看端口情况

netstat -nat | awk '{print $5}' | awk -F: '{print $2}' | sort|uniq -c|sort -r

查看进程在那个端口堆积最多，比如80(http) 3306(mysql) 6379(redis)常用的端口

查看IP最多

netstat -na|grep ESTABLISHED|awk '{print $5}' |awk -F: '{print $1}' |sort|uniq -c|sort -r

查看那个IP访问服务器最多，从而判别是否被某个IP攻击（或者黄牛）

phptrace

利用phptrace排查php-fpm阻塞原因

phptrace -p $(pgrep -n php-fpm) 2>&1 | awk '$NF > 0.0001'

其中利用awk过滤执行时间大于0.0001S的函数

监控LOAD报警

监控记录LOAD负载过高情况下的ps情况，添加shell脚本ps.sh，python邮件报警必须安装email模块pip install email，下载脚本

#/bin/bash

LOAD=$(awk '{print $1}' /proc/loadavg)
CPUNUM=$(grep -c processor /proc/cpuinfo)


if [ $(echo "$LOAD > $CPUNUM" | bc) = 1 ]; then
    RESULT=$(ps -eo pcpu,pmem,user,args | awk '$1 > 0' | sort -nr)
    if [ -n "$RESULT" ]; then
        logfile=/var/log/ps/ps.$(date +"%Y%m%d%H%M").log
        echo "$RESULT" > ${logfile}
        /usr/local/scripts/sendmail.py -t caixudong@verystar.cn -s '服务器PS报警' -f $logfile
    fi
fi

加入定时任务执行

#监控CPU使用量，详情请看火丁笔记http://huoding.com/2013/01/30/225
*/1 * * * * /usr/local/scripts/ps.sh >/dev/null 2>&1

定时清理LOG

touch /etc/logrotate.d/pslog
vim /etc/logrotate.d/pslog

/var/log/ps/*.log {
        daily
        missingok
        rotate 7
        compress
        delaycompress
        notifempty
        sharedscripts
        prerotate
               #转储存之前合并小文件
                for i in ps.*.log; do echo $i >> ps.log; cat $i >> ps.log; done
        endscript
        postrotate
               #转储存之后执行删除小文件
                rm -rf ps.*.log
        endscript
}
